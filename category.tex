\section{圏の定義}

\begin{Definition}[圏の定義] \mbox{}

\begin{description}
\item [Object] \mbox{}

圏の対象

\item [Arrow/Morphism] \mbox{}

圏の射

\item [dom(始域), cod(終域)] \mbox{}

各射にたいしてdom(f)とcod(f)が割り当てられている
\begin{align}
dom(f)=A \\
cod(f)=B
\end{align}

のとき
$f:A \to B$
または、
$A \stackrel{f}{\to} B$
と表現し、fはAからBへの射であると言う。
AからBへの射全体を $C(A,B)$と表現する。
可換図で表現すると以下のとおりとなる

$$
\xymatrix{
  A \ar[r]^f & B \\
}
$$

圏Cの対象AからBへの射の集合を次の様にかく
\[
 Hom_{Sets^{C}}(A,B) = Hom_{C}(A,B) = Hom(A,B) = C(A,B) = (A,B) = (A,B)_{C}
\]
この記述は全て同じ意味である



\item [composition(射の合成)] \mbox{}

$cod(f) = dom(g)$ならば

%%A \stackrel{f}{\to} B \stackrel{g}{\to} C
$$
\begin{CD}
A @>f >> B @>g >>C \\
\end{CD}
$$

図の様に$ f : A \to B $と $ f : B \to C $の二つの射が合成出来る。
射の合成を  $ g \circ f $ または、 $ f ; g $ と表現する

可換図で書くと次のとおりである

$$
\xymatrix{
 A \ar[r]^f \ar[dr]_{f;g}
  & B \ar[d]^g \\
  & C }
$$



\end{description}

\end{Definition}

\section{圏の公理}
\begin{description}


\item [合成] \mbox{}

$ g \circ f $  が定義されていれば、
$ dom(g \circ f) = dom(f) $ かつ $ cod(g \circ f) = cod(g) $
である
可換図で表現すると次の形となる

$$
\xymatrix{
 A \ar[r]^f \ar[dr]_{f;g}
  & B \ar[d]^g \\
  & C }
$$

\item [結合則] \mbox{}

任意の射 $ f : A \to B 、f : B \to C、 h : C \to D $ が存在したときに、
$ \circ $ は結合法則を満たす
\[
 h \circ (g \circ f) = (h \circ g) \circ f
\]

また、$ f : A \to B  $と $ f : B \to C $となる射と $ h : A \to C $となる射が存在し

$$
\xymatrix{
   A \ar@{->}[r]^{f}\ar@{->}[d]_{h}  & B \ar@{->}[dl]^{g}\\
   C  & \\
}
$$
となるhが存在する場合、
hと$ f \circ g $ は可換であるとする





\item [identity (恒等射)] \mbox{}

各対象Bについて恒等射 $ id_B: B \to B $ が定義されている。
恒等射は $ f : A \to B $, $ g : B \to C $ に対して
$ id_B \circ g = g $ かつ  $ f \circ id_B = f $
が成り立つ。
これは単位元律という。

恒等射は次の様にも記述する。
$ 1_{f} $


$$
\xymatrix{
 A \ar@{->}[r]^{g} \ar@{->}[dr]_{g} & B \ar@{->}[r]^{f}  \ar@{->}[d]^{id_B} & C \\
 & B \ar@{->}[ur]_{f} }
$$




\end{description}

\section{関手}
\begin{Definition}[関手]
圏A,Bが存在し、

$$\xymatrix{A \ar@{->}[r]^{F} & B}$$

圏Aの任意の対象と任意の射をそれぞれ圏Bを移す写像Fを関手と
定義する。関手であるとは可換図で以下のように書く

$$
\xymatrix{
F(A_1) \ar@{->}[d]_{f} \ar@{=>}[r]_{F} & F(A_1) = B_1 \ar@{->}[d]_{F(f)}  \\
F(A_2)                 \ar@{=>}[r]_{F} & F(A_2) = B_2
}
$$

関手であるとは以下の二つの条件を満たしていることと定義する。

合成
\[
 F(g \circ f) = F(g) \circ^{\prime} F(f)
\]


恒等射
\[
 F(id_A) = id^{\prime}_F(A)
\]


$$
\xymatrix{
F(A_1) \ar@{->}[d]_{f} \ar@{=>}[r]_{F} & F(A_1) = B_1 \ar@{->}[d]_{F(f)}  \\
F(A_2) \ar@{->}[d]_{g} \ar@{=>}[r]_{F} & F(A_2) = B_2 \ar@{->}[d]_{F(g)}  \\
F(A_3)                 \ar@{=>}[r]_{F} & F(A_3) = B_3
}
$$
\end{Definition}



\begin{align*}

G : C \to D
\end{align*}

\[

\section{自然変換}
\begin{Definition}[自然変換]
圏C,Dが存在し、CからDへの関手F,Gが存在する事
\[
 F,G : C \to D
\]
この二つの関手F,Gに $ \mu_{A} $ の様なFからGへ移す射が存在した場合これを自然変換と呼
 ぶ

各対象を移す
\[
 \mu_A : F(A) \to G(A)
\]
各写像を移す。この時合成則を満たす事
\[
 G(f) \circ \mu_A = \mu_B \circ F(f)
\]

$$
\begin{CD}
F(A) @>\mu_A >> G(A) \\
@V F(f) VV  @V G(f)VV \\
F(B) @>\mu_B >> G(B)
\end{CD}
$$

\end{Definition}
